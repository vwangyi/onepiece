## 创建数组

```js
// 构造函数Array 字面量[]  Array.of()
new Array(1，2，3)
[1, 2, 3]
new Array(3).fill(0).map((_,i) => i+1)
Array.of() // []
Array.of(1); // [1]
Array.of(1, 2, 3); // [1, 2, 3]
// Array.of() 和 new Array的区别
// 1 一个参数时 new Array表示的是 数组长度  Array.of 表示数组项
// Array.of创建数组的性能 比 new Array 略低 但可以忽略不计
// es6新增的Array.of 解决了 new Array 的歧义
```

## 类数组转真数组

```js
// Array.from() Array.fromAsync() 可以将 可迭代对象 或 类数组对象 转为 浅拷贝后的数组实例。
[...new Set([1, 2, 3, 4])];
Array.from({ length: 3 });
Array.from("hello");
Array.from({
  *[Symbol.iterator]() {
    yield 1;
    yield 2;
    yield 3;
  }
}); // [1,2,3]
// 类数组：只有长度和下标
```

## 数组遍历

```js
// for循环 for-of forEach

for (let i = 0; i < arr.length; i++) {
  console.log(arr[i]); // i是下标 arr[i]是元素 完美循环 没有缺点
}
for (let i of [1, 2, 3]) {
  console.log(i); // item是元素  缺点没有索引
}
[1, 2, 3].forEach((item, index, arr) => {
  // forEach不支持break continue return
  // forEach只能throw new Error()抛出错误
  // 用try catch捕获错误 来终止循环
  // 但一般不会手动终止forEach循环 如需要终止循环请不要用forEach
});
```

## 数组映射

```js
// map方法 把原数组 映射为 浅拷贝的新数组
[1, 2, 3].map(item => 1); // [1,1,1]
```

## 数组过滤 filter

```js
// filter方法 把原数组 进行筛选 项数可能会变少
[1, 2, 3].filter(item => item > 1); // [2,3]
[0, 1].filter(Boolean); // [1]
```

## 数组排序

```js
// - sort 和 toSorted 的区别是 toSorted不影响原数组 返回新数组
// - reverse 和 toReversed 的区别是  toReversed 不影响原数组 返回新数组

[2, 1].sort((a, b) => a - b); // 升序 影响原数组
[2, 1].sort((a, b) => b - a); // 降序 影响原数组

["a", "v", "b"].sort(); // 按UTF-16排序 字母排序
["张三", "李四"].sort((a, b) => a.localeCompare(b, "zh-CN")); // 国际化多语言排序

[2, 1].reverse(); // 反转顺序
[2, 1].toReversed(); // 反转顺序
```

## 数组去重 reduce

```js
const unique = [...new Set([1, 1, 1, 2, 3, 4, 5, 5])]; // [1,2,3,4,5] 只能处理原始类型
// 可以处理 对象数组
const unique = Object.values(
  [{ id: 1 }, { id: 1 }].reduce((prev, curr) => {
    prev[curr.id] = curr;
    return prev; // 返回给下一次使用
  }, {})
);
// 数组reduce是万能的 数组的操作都可以用reduce  累加 计算等等
```

## 数组增删改查

```js

[1,2,3].unshift(1,2,3); // 增头部 返回总长度
[1,2,3].pop(); // 删尾部 返回删除元素

arr.length = 0; // 清空数组 但不会修改地址
arr = []; // 赋值新数组


// 对数据 改 操作
arr[下标] = 新值 // 修改下标对应位置的元素
arr.with(下标, 新值)； // 返回新数组 不影响原数组


arr.length // 查数组长度

arr.findIndex() arr.indexOf() arr.lastIndexOf()// 查数组下标 找不到返回-1
```

## 数组扁平化

```js
// 对数据 扁平化处理
arr.flat(); // 默认扁平一层

```

## 数组切片 slice

```js
arr.slice(-1)[0]; // 获取最后一个元素 浅拷贝
arr.slice(-2)[0]; // 查倒数第二个元素 浅拷贝

/* 切片操作 */
[1, 2, 3].slice(开始索引, 结束索引);
// 左闭右开 包含左 不包含右 结束索引不写 默认到最后 返回新数组 浅拷贝
// 支持反向索引 -1是最后一个元素
[(1, 2, 3)].slice(-1)[0];
["a", "b", "c", "d", "e"].slice(2, -2);
```

## 数组连接 join

```js
// 数组join连接转字符串
arr.join(); // 默认用逗号连接
arr.join(""); // 用空字符串连接
arr.join("-"); // 用-连接
```
