# 依赖注入 provide inject

- https://cn.vuejs.org/guide/components/provide-inject.html 

## vue3 setup写法
- 祖先组件
```vue  
<script setup> 
import { ref, provide, readonly } from 'vue'
const location = ref(123); 
function updateLocation(value) {
  location.value = value;
} 
provide(/* 自定义注入名 */ 'message', /* 值 */ 'hello!')  
// 传变量 就是向下传数据  传函数 就是向上传数据
provide('location', { 
  location: readonly(location), // 不想让后代修改 就用readonly包裹
  updateLocation // 如果后代需要修改 建议祖先组件提供修改函数 
})
</script>

```
- 后代组件
```vue
<script setup> 
import { ref, inject } from 'vue'
const { location, updateLocation } = inject('location');

import { inject } from 'vue' 
const message = inject('message', '这是默认值')// 第二个参数是默认值
const message = inject('key', () => new ExpensiveClass(), true); // 第三个参数传true 表示第二个参数是一个函数

// 建议尽可能将任何对响应式状态的变更都保持在供给方组件中。

// inject是响应式的 可以监听 
</script>
```

## vue2写法 
- 祖先组件
```vue 
<script>
export default {
  // 和data一样 不能是箭头函数
  provide: function() {
    return {
      gethaha: '123123'// 向后代组件传递 gethaha参数
    }
  }
}
</script> 
<script>
import { createApp } from 'vue' 
const app = createApp({}) 
// 在应用级别提供的数据在该应用内的所有组件中都可以注入。这在你编写插件时会特别有用，因为插件一般都不会使用组件形式来提供值。
app.provide(/* 注入名 */ 'message', /* 值 */ 'hello!');
</script>
```  
- 后代组件
```vue
<script>
  
export default {
  inject: ['gethaha'],
  methods: {
     handle() {
        console.log(this.gethaha)
     }
  }
}
 
</script>
```
