```sql 
select * from `表名` where 条件;  -- *表示查询表中所有字段
```

```sql
/*
   思路

   2.确定使用哪种连接查询？ (7种) （本质3种 扩展7种）

-- inner join 内连接
select s.studentNo, studentName, SubjectNo, StudentResult   -- 因为两张表都有studentNo字段 所以起别名 用别名. 来指定是哪个表的字段

from student as s right join result as r    -- 表名 as 别名    这里的as 可以省略
on s.studentNo = r.studentNo;   -- where 可以 改为 on 两者效果一样


-- left join  左连接
select s.studentNo, studentName, SubjectNo, StudentResult   -- 因为两张表都有studentNo字段 所以起别名 用别名. 来指定是哪个表的字段
from student as s    -- 给student表 起别名 s    这里的as 可以省略
left join result r  -- 给result表 起别名 r   这里的as 可以省略
on s.studentNo = r.studentNo;   -- where 可以 改为 on 两者效果一样


操作                  描述

right join  右连接
```

## select 查询完整语法
```sql
select [all | distinct] -- distinct 去重
{ * | table.* | [table.field1[as alias1][,table.field2[as alias2]][,...]]}
from 表名 [as 别名]
    [inner | left | right] join 表名 [as 别名] [on 条件]  -- 连接查询
    [where 条件] -- 指定结果需要满足的条件
    [group by 字段] -- 指定结果需满足的条件
    [having 条件] -- 过滤分组的记录 必须满足的次要条件
    [order by 字段 [asc | desc]] -- 指定查询记录按一个或多个条件排序
    [limit m,n] -- 指定查询记录的范围 从哪条 到 哪条

-- [] 表示可选  {} 表示必选
```



## 子查询
```sql
-- 子查询: 查询结果作为另一个查询的条件
-- 1. 查询 数据库结构-1 的所有考试结果 (学号 科目编号 成绩) 降序排列

-- 方式1  使用连接查询
select `studentNo`, r.`subjectNo`, `studentResult`
from result r inner join `subject` sub
on r.`subjectNo` = sub.`subjectNo`
where sub.`subjectName` = '数据库结构-1'
order by `studentResult` desc;


-- 方式2 使用子查询 (先里后外)

select `studentNo`, `subjectNo`, `studentResult`
from result
where `subjectNo` = (
  -- 先用子查询 查询出 subject 的subjectNo字段   等于  result的subjectNo
  select `subjectNo` from `subject` 
  where `subjectName` = '数据库结构-1'
)

-- 如果查询字段 多个表都有这个字段 就需要as给表取别名 通过别名指定字段了
 


-- 查询课程为 高等数学-2 且分数不小于 80 的同学的学号和姓名
-- 方式 1  使用的是连接查询 (代码多 好理解)
select s.`studentNo`, `studentName`
from student s inner join result r on s.`studentNo` = r.`studentNo`
inner join `subject` sub on r.`subjectNo` = sub.`subjectNo`
where sub.`subjectName` = '高等数学-2' and r.`studentResult` >= 80;

-- 方式 2  连接查询和子查询结合
select distinct s.`studentNo`, `studentName`
from student s inner join result r on s.`studentNo` = r.`studentNo`
where `studentResult` >= 80 and `subjectNo` = (
  select `subjectNo` from `subject` where `subjectName` = '高等数学-2'
);

-- 方式 3  子查询 和 嵌套查询 (先里后外) （代码最少 难理解）
select studentNo, studentName from student where studentNo in (
  select `studentNo` from result where `studentResult` >= 80 and `subjectNo` = (
    select `subjectNo` from `subject` where `subjectName` = '高等数学-2'
  )
);
```

### 问：join联表查询
+ 是什么：联表查询就是 两张表连起来查询，常见的有 left join 、right join、inner join
+ left join 就是 返回左表的所有字段
+ right join 是 返回右表的所有字段
+ inner join 是 返回两个表的交集字段
+ 应用场景：需要查询的字段 来自多张表 

```sql

-- 查询缺考的同学
select s.studentNo, studentName, SubjectNo, StudentResult   
from stdudent as s 
left join result as r on s.studentNo = r.studentNo 
where StudentResult is null   -- 缺考就是没成绩 那就是null

 
-- join（联接的表）on（判断的条件） 联接查询  （join 和 on结合 联接查询）
-- where   等值查询。                        （ where 是 等值查询 ）
 
// 目前有3张表：学生表student 成绩表result 科目表subject
// 查询参加考试的同学信息 查询字段有: 学号 学生姓名 科目名 分数

-- 确定最终查询的结果字段
select s.studentNo, studentName, SubjectName, `StudentResult`
from student as s -- 一张表
-- 使用 right join 连接成绩表。查询条件是 学号相同的
right join result as r on r.studentNo = s.studentNo
-- 使用 inner join 连接 科目表 查询条件是 科目名相同的
inner join `subject` as sub on r.SubjectNo = sub.SubjectNo

 
```
