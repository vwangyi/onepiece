// 实际场景：无限滚动分页
async function* createPaginatedFetcher(apiEndpoint) {
  let page = 1;
  let hasMore = true;
  
  while (hasMore) {
    // 实际API调用
    const response = await fetch(`${apiEndpoint}?page=${page}`);
    const data = await response.json();
    
    if (data.items.length === 0) {
      hasMore = false;
      break;
    }
    
    // 每次yield一页数据
    yield data.items;
    page++;
  }
}

// 使用：优雅的无限滚动
async function handleInfiniteScroll() {
  const itemFetcher = createPaginatedFetcher('/api/products');
  
  for await (const pageItems of itemFetcher) {
    // 渲染当前页
    renderItems(pageItems);
    
    // 检查是否需要继续加载
    if (shouldStopLoading()) {
      break;
    }
  }
}

function* characterStateMachine() {
  while (true) {
    // 空闲状态
    yield { state: 'idle', animation: 'stand', duration: 2000 };
    
    // 巡逻状态
    const patrolPoints = [[0, 0], [100, 0], [100, 100], [0, 100]];
    for (const point of patrolPoints) {
      yield { 
        state: 'patrol', 
        action: 'move', 
        target: point,
        duration: 1000 
      };
      yield { state: 'look_around', duration: 500 };
    }
    
    // 发现敌人
    yield { state: 'alert', animation: 'draw_weapon', duration: 300 };
    
    // 攻击序列
    for (let i = 0; i < 3; i++) {
      yield { state: 'attack', type: 'swing', damage: 10 };
      yield { state: 'cooldown', duration: 400 };
    }
  }
}

// 使用
const enemyAI = characterStateMachine();

function gameLoop() {
  const currentAction = enemyAI.next().value;
  
  switch (currentAction.state) {
    case 'idle':
      playAnimation(currentAction.animation);
      break;
    case 'patrol':
      moveCharacter(currentAction.target);
      break;
    case 'attack':
      calculateDamage(currentAction.damage);
      break;
  }
  
  // 设置下一次状态切换
  setTimeout(gameLoop, currentAction.duration);
}

## 🔄 **3. 异步流程控制（复杂工作流）**

```javascript
// 实际场景：多步骤表单提交
function* multiStepForm() {
  try {
    // 步骤1: 验证表单
    console.log('步骤1: 验证表单...');
    const isValid = yield validateForm();
    
    if (!isValid) {
      throw new Error('表单验证失败');
    }
    
    // 步骤2: 上传文件
    console.log('步骤2: 上传文件...');
    const fileUrls = yield uploadFiles();
    
    // 步骤3: 创建草稿
    console.log('步骤3: 创建草稿...');
    const draftId = yield createDraft(fileUrls);
    
    // 步骤4: 发送通知
    console.log('步骤4: 发送通知...');
    yield sendNotifications(draftId);
    
    // 步骤5: 最终提交
    console.log('步骤5: 最终提交...');
    const result = yield finalSubmit(draftId);
    
    return { success: true, data: result };
    
  } catch (error) {
    // 统一错误处理
    console.error('流程失败:', error);
    yield rollbackChanges(); // 回滚
    return { success: false, error };
  }
}

// 运行器
async function runWorkflow(generator) {
  const gen = generator();
  let result = gen.next();
  
  while (!result.done) {
    try {
      // 等待异步操作完成
      const value = await result.value;
      result = gen.next(value);
    } catch (error) {
      result = gen.throw(error);
    }
  }
  
  return result.value;
}

// 使用
runWorkflow(multiStepForm).then(result => {
  if (result.success) {
    showSuccessMessage();
  } else {
    showErrorMessage(result.error);
  }
});

## 🕹️ **6. 撤消/重做功能（状态快照）**

```javascript
// 实际场景：编辑器状态历史
function* createHistoryManager() {
  const history = [];
  let currentIndex = -1;
  
  while (true) {
    const action = yield { 
      canUndo: currentIndex > 0,
      canRedo: currentIndex < history.length - 1,
      history 
    };
    
    switch (action.type) {
      case 'SAVE_STATE':
        // 清除当前索引之后的历史
        history.splice(currentIndex + 1);
        history.push(action.state);
        currentIndex = history.length - 1;
        break;
        
      case 'UNDO':
        if (currentIndex > 0) {
          currentIndex--;
          return history[currentIndex]; // 返回历史状态
        }
        break;
        
      case 'REDO':
        if (currentIndex < history.length - 1) {
          currentIndex++;
          return history[currentIndex];
        }
        break;
        
      case 'CLEAR':
        history.length = 0;
        currentIndex = -1;
        break;
    }
  }
}

// 使用：文本编辑器
class TextEditor {
  constructor() {
    this.history = createHistoryManager();
    this.history.next(); // 初始化
    this.content = '';
  }
  
  type(text) {
    this.content += text;
    this.saveState();
  }
  
  saveState() {
    this.history.next({
      type: 'SAVE_STATE',
      state: { content: this.content, timestamp: Date.now() }
    });
  }
  
  undo() {
    const state = this.history.next({ type: 'UNDO' }).value;
    if (state) {
      this.content = state.content;
    }
  }
  
  redo() {
    const state = this.history.next({ type: 'REDO' }).value;
    if (state) {
      this.content = state.content;
    }
  }
}

## 🔌 **7. WebSocket/SSE 消息流处理**

// 实际场景：实时聊天消息处理
function* createMessageProcessor() {
  const buffers = {
    text: [],
    image: [],
    file: []
  };
  
  let messageCount = 0;
  
  while (true) {
    const message = yield; // 等待消息
    
    switch (message.type) {
      case 'TEXT':
        buffers.text.push(message);
        
        // 每10条文本消息批量处理一次
        if (buffers.text.length >= 10) {
          yield { 
            type: 'BATCH_TEXT',
            messages: buffers.text.splice(0, 10),
            count: messageCount += 10
          };
        }
        break;
        
      case 'IMAGE':
        buffers.image.push(message);
        
        // 图片消息立即处理
        yield {
          type: 'PROCESS_IMAGE',
          message,
          count: ++messageCount
        };
        break;
        
      case 'TYPING':
        // 实时状态，不计数
        yield {
          type: 'USER_TYPING',
          userId: message.userId
        };
        break;
        
      case 'BATCH_PROCESS':
        // 手动触发批量处理
        if (buffers.text.length > 0) {
          yield {
            type: 'FLUSH_BUFFER',
            messages: buffers.text.splice(0),
            count: messageCount += buffers.text.length
          };
        }
        break;
    }
  }
}

// 使用
const processor = createMessageProcessor();
processor.next(); // 启动

// WebSocket 事件处理
websocket.onmessage = (event) => {
  const message = JSON.parse(event.data);
  const result = processor.next(message);
  
  if (!result.done) {
    handleProcessedMessage(result.value);
  }
};

## 🌐 **8. 路由中间件系统（Koa/Express风格）**

```javascript
// 实际场景：自定义中间件管道
function* createMiddlewarePipeline() {
  const request = yield; // 接收初始请求
  
  // 中间件1: 日志记录
  console.log(`[${new Date().toISOString()}] ${request.method} ${request.url}`);
  let context = { ...request, startTime: Date.now() };
  
  // 中间件2: 认证检查
  const authResult = yield checkAuth(context);
  if (!authResult.valid) {
    return { status: 401, error: '未授权' };
  }
  context.user = authResult.user;
  
  // 中间件3: 参数验证
  const validation = yield validateParams(context);
  if (!validation.valid) {
    return { status: 400, error: validation.errors };
  }
  context.params = validation.params;
  
  // 中间件4: 业务处理
  const data = yield handleBusinessLogic(context);
  
  // 中间件5: 响应格式化
  const response = yield formatResponse(data);
  
  // 中间件6: 性能监控
  const duration = Date.now() - context.startTime;
  console.log(`请求耗时: ${duration}ms`);
  
  return response;
}

// 运行器
async function handleRequest(request) {
  const pipeline = createMiddlewarePipeline();
  pipeline.next(request); // 传入初始请求
  
  let result = pipeline.next();
  while (!result.done) {
    const middlewareResult = await result.value;
    result = pipeline.next(middlewareResult);
  }
  
  return result.value;
}

| **中间件系统** | 可组合的请求处理链 | 自定义HTTP框架 |

## 🎯 **何时选择生成器？**

选择生成器函数的**最佳时机**：

1. **需要保持状态**：函数需要记住上次执行的位置
2. **惰性求值**：数据需要时才生成，节省内存
3. **复杂流程**：多步骤且有条件分支的流程
4. **异步序列**：多个异步操作需要顺序执行
5. **数据流处理**：数据需要经过多个处理阶段

## ⚠️ **注意事项**

```javascript
// 不适合生成器的场景：
// 1. 简单的异步操作 → 用 async/await
async function simpleFetch() {
  const data = await fetch(url);
  return data;
}

// 2. 一次性计算 → 用普通函数
function calculateTotal(items) {
  return items.reduce((sum, item) => sum + item.price, 0);
}

// 3. 事件监听器 → 用回调或EventEmitter
element.addEventListener('click', () => {
  // 处理点击
});
```

**生成器函数是JavaScript中一个被低估的强大工具**。虽然async/await解决了大部分异步问题，但在需要**精细控制执行流程**、**复杂状态管理**或**惰性数据生成**的场景下，生成器仍然是无可替代的选择！
