
# 单向链表 
- https://leetcode.cn/problems/design-linked-list/description/ 链表初始化
- 封装单向链表 
```ts
class ListNode {
    val: number;
    next: ListNode | null;
    
    constructor(val?: number, next?: ListNode | null) {
        this.val = (val === undefined ? 0 : val);
        this.next = (next === undefined ? null : next);
    }
}

class MyLinkedList {
    private size: number;
    private head: ListNode;

    constructor() {
        this.size = 0;
        this.head = new ListNode(0); // 虚拟头节点
    }

    // 获取链表中第 index 个节点的值
    get(index: number): number {
        if (index < 0 || index >= this.size) {
            return -1;
        }
        let cur: ListNode | null = this.head;
        for (let i = 0; i <= index; i++) {
            cur = cur!.next;
        }
        return cur!.val;
    }

    // 在链表头部添加节点
    addAtHead(val: number): void {
        this.addAtIndex(0, val);
    }

    // 在链表尾部添加节点
    addAtTail(val: number): void {
        this.addAtIndex(this.size, val);
    }

    // 在链表中第 index 个节点前添加节点
    addAtIndex(index: number, val: number): void {
        if (index > this.size) {
            return;
        }
        index = Math.max(0, index);
        this.size++;
        
        let pred: ListNode | null = this.head;
        for (let i = 0; i < index; i++) {
            pred = pred!.next;
        }
        
        const toAdd = new ListNode(val);
        toAdd.next = pred!.next;
        pred!.next = toAdd;
    }

    // 删除链表中第 index 个节点
    deleteAtIndex(index: number): void {
        if (index < 0 || index >= this.size) {
            return;
        }
        this.size--;
        
        let pred: ListNode | null = this.head;
        for (let i = 0; i < index; i++) {
            pred = pred!.next;
        }
        pred!.next = pred!.next!.next;
    }

    // 可选：添加一个 toString 方法方便查看链表内容
    toString(): string {
        const values: number[] = [];
        let current: ListNode | null = this.head.next;
        
        while (current !== null) {
            values.push(current.val);
            current = current.next;
        }
        
        return `MyLinkedList: [${values.join(' -> ')}] (size: ${this.size})`;
    }
}

// 测试代码
const linkedList = new MyLinkedList();
console.log('空链表:', linkedList.toString());

// 尾部添加节点
linkedList.addAtTail(1);    // 链表: 1
linkedList.addAtTail(2);    // 链表: 1 -> 2
linkedList.addAtTail(3);    // 链表: 1 -> 2 -> 3

console.log('添加三个节点后:', linkedList.toString());
console.log('当前链表长度:', linkedList['size']); // 注意：size 是私有属性

// 测试获取
console.log('获取索引0:', linkedList.get(0)); // 1
console.log('获取索引1:', linkedList.get(1)); // 2
console.log('获取索引2:', linkedList.get(2)); // 3
console.log('获取无效索引:', linkedList.get(3)); // -1

// 头部添加
linkedList.addAtHead(0);
console.log('头部添加0后:', linkedList.toString()); // 0 -> 1 -> 2 -> 3

// 在指定位置添加
linkedList.addAtIndex(2, 99);
console.log('在索引2添加99后:', linkedList.toString()); // 0 -> 1 -> 99 -> 2 -> 3

// 删除
linkedList.deleteAtIndex(2);
console.log('删除索引2后:', linkedList.toString()); // 0 -> 1 -> 2 -> 3

// 查看链表内部结构（私有属性可以通过这种方式访问，但不推荐）
console.log('链表对象:', linkedList);
```
