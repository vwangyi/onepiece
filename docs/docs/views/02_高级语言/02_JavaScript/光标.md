

我来详细解释 `window.getSelection()` 和相关的 API：

## 1. Selection 对象基础
`window.getSelection()` 返回一个 **Selection 对象**，代表用户选择的文本范围或光标位置。

```javascript
const selection = window.getSelection();
```

## 2. Selection 对象的主要属性
### 2.1 `rangeCount` - 选区数量
```javascript
selection.rangeCount
```

+ **含义**：当前选区中包含的 Range 对象的数量
+ **值**：整数，通常为 0 或 1
    - `0`：没有选择任何内容
    - `1`：一个连续的选区（最常见）
    - `>1`：多个不连续的选区（在某些编辑器中可能，如 Firefox 的多选区编辑）

### 2.2 `isCollapsed` - 是否折叠
```javascript
selection.isCollapsed
```

+ **含义**：选区起始点和结束点是否相同（即光标位置，而非选择范围）
+ **值**：布尔值
    - `true`：光标位置（没有选择文本）
    - `false`：选择了文本范围

### 2.3 `anchorNode` 和 `focusNode`
```javascript
selection.anchorNode    // 选区开始的节点
selection.focusNode     // 选区结束的节点
selection.anchorOffset  // 在 anchorNode 中的偏移量
selection.focusOffset   // 在 focusNode 中的偏移量
```

#### 选择方向很重要：
```javascript
// 用户从左到右选择："hello world"
// anchor在"h"前面，focus在"d"后面

// 用户从右到左选择："hello world"  
// anchor在"d"后面，focus在"h"前面
```

### 2.4 `type` - 选区类型
```javascript
selection.type
```

+ **含义**：选区类型
+ **值**：
    - `"None"`：没有选择
    - `"Caret"`：光标位置（折叠选区）
    - `"Range"`：选择了文本范围

## 3. Range 对象详解
`selection.getRangeAt(index)` 返回 Range 对象：

```javascript
if (selection.rangeCount > 0) {
  const range = selection.getRangeAt(0);
  
  // Range 的关键属性
  range.startContainer    // 起始节点
  range.startOffset      // 起始偏移量
  range.endContainer     // 结束节点
  range.endOffset       // 结束偏移量
  range.collapsed       // 是否折叠
  
  // 常用方法
  range.getBoundingClientRect()  // 获取位置和尺寸
  range.getClientRects()         // 获取所有矩形（多行选择）
}
```

## 4. 详细示例和解释
### 示例 1：基本使用
```javascript
const selection = window.getSelection();

console.log('rangeCount:', selection.rangeCount);  // 0 或 1
console.log('isCollapsed:', selection.isCollapsed); // 是否只是光标

if (selection.rangeCount > 0) {
  const range = selection.getRangeAt(0);
  
  console.log('开始节点:', range.startContainer);
  console.log('开始位置:', range.startOffset);
  console.log('结束节点:', range.endContainer); 
  console.log('结束位置:', range.endOffset);
  console.log('是否折叠:', range.collapsed);
  
  // 获取选中的文本
  console.log('选中文本:', range.toString());
  
  // 获取视觉位置
  const rect = range.getBoundingClientRect();
  console.log('位置:', rect);
}
```

### 示例 3：处理不同类型的选择
```javascript
function analyzeSelection() {
  const sel = window.getSelection();
  
  if (sel.rangeCount === 0) {
    console.log('没有选择内容');
    return;
  }
  
  // 获取主选区
  const range = sel.getRangeAt(0);
  
  // 判断选择状态
  if (sel.isCollapsed || range.collapsed) {
    console.log('这是光标位置（折叠选区）');
    console.log('光标在:', getCursorContext(range));
  } else {
    console.log('选择了文本范围');
    console.log('选择的内容:', range.toString());
    console.log('选择长度:', range.toString().length);
  }
  
  // 获取选择方向
  console.log('选择方向:', getSelectionDirection(sel));
}

function getCursorContext(range) {
  // 获取光标前后的文本
  const node = range.startContainer;
  const offset = range.startOffset;
  
  if (node.nodeType === Node.TEXT_NODE) {
    const text = node.textContent;
    const before = text.substring(0, offset);
    const after = text.substring(offset);
    
    return {
      node: node,
      textBefore: before,
      textAfter: after,
      position: offset
    };
  }
  
  return null;
}

function getSelectionDirection(selection) {
  // 判断选择方向（从左到右或从右到左）
  const range = selection.getRangeAt(0);
  
  // 比较锚点和焦点的位置
  const position = range.compareBoundaryPoints(
    Range.START_TO_START, 
    range
  );
  
  if (selection.anchorOffset < selection.focusOffset) {
    return '从左到右';
  } else if (selection.anchorOffset > selection.focusOffset) {
    return '从右到左';
  } else {
    return '未知';
  }
}
```

## 5. Selection 的方法
### 5.1 操作方法
```javascript
// 添加一个Range到选区
selection.addRange(range);

// 从选区移除一个Range
selection.removeRange(range);

// 移除所有Range
selection.removeAllRanges();

// 删除文档中的选中内容
selection.deleteFromDocument();

// 选中指定节点的内容
selection.selectAllChildren(element);
```

### 5.2 修改方法
```javascript
// 扩展选区到指定节点
selection.extend(node, offset);

// 设置选区为单个Range
selection.setBaseAndExtent(anchorNode, anchorOffset, focusNode, focusOffset);

// 包含指定节点
selection.containsNode(node, partlyContained);

// 转换为字符串
selection.toString();
```

## 6. 实际应用场景
### 场景 1：保存和恢复光标
```javascript
class CursorManager {
  save() {
    const sel = window.getSelection();
    if (sel.rangeCount === 0) return null;
    
    const range = sel.getRangeAt(0);
    
    return {
      // 保存完整的Range信息
      startContainer: range.startContainer,
      startOffset: range.startOffset,
      endContainer: range.endContainer,
      endOffset: range.endOffset,
      
      // 保存视觉位置（用于恢复时检查）
      rect: range.getBoundingClientRect(),
      
      // 保存文本上下文（用于容错）
      textBefore: this.getTextBefore(range, 50),
      textAfter: this.getTextAfter(range, 50)
    };
  }
  
  restore(cursorInfo) {
    const sel = window.getSelection();
    sel.removeAllRanges();
    
    const range = document.createRange();
    
    try {
      // 尝试直接恢复
      range.setStart(cursorInfo.startContainer, cursorInfo.startOffset);
      range.setEnd(cursorInfo.endContainer, cursorInfo.endOffset);
    } catch (e) {
      // 如果节点不存在，使用文本上下文恢复
      this.restoreByContext(cursorInfo);
      return;
    }
    
    sel.addRange(range);
  }
}
```

### 场景 2：在 contenteditable 中插入内容
```javascript
function insertAtCursor(text) {
  const selection = window.getSelection();
  
  if (selection.rangeCount === 0) {
    // 没有选区，插入到末尾
    editor.innerHTML += text;
    return;
  }
  
  const range = selection.getRangeAt(0);
  
  // 删除当前选中的内容（如果有）
  range.deleteContents();
  
  // 创建文本节点并插入
  const textNode = document.createTextNode(text);
  range.insertNode(textNode);
  
  // 移动光标到插入的文本后面
  range.setStartAfter(textNode);
  range.collapse(true);
  
  // 更新选区
  selection.removeAllRanges();
  selection.addRange(range);
}
```

### 场景 3：获取光标所在位置信息
```javascript
function getCursorInfo() {
  const sel = window.getSelection();
  
  if (sel.rangeCount === 0 || sel.isCollapsed) {
    const range = sel.getRangeAt(0);
    
    return {
      type: 'cursor',
      node: range.startContainer,
      offset: range.startOffset,
      textNode: range.startContainer.nodeType === Node.TEXT_NODE 
        ? range.startContainer 
        : null,
      
      // 获取所在行
      lineInfo: getCurrentLineInfo(range),
      
      // 获取在文本中的位置
      positionInText: getPositionInText(range),
      
      // 视觉位置
      rect: range.getBoundingClientRect()
    };
  } else {
    // 文本选择状态
    return {
      type: 'selection',
      text: sel.toString(),
      length: sel.toString().length,
      range: sel.getRangeAt(0)
    };
  }
}

function getCurrentLineInfo(range) {
  // 向上查找最近的块级元素
  let element = range.startContainer;
  
  if (element.nodeType === Node.TEXT_NODE) {
    element = element.parentElement;
  }
  
  while (element && !isBlockElement(element)) {
    element = element.parentElement;
  }
  
  return element ? {
    element: element,
    tagName: element.tagName,
    className: element.className,
    text: element.textContent
  } : null;
}

function isBlockElement(el) {
  const blockTags = ['P', 'DIV', 'H1', 'H2', 'H3', 'H4', 'H5', 'H6', 'LI', 'PRE'];
  return blockTags.includes(el.tagName);
}
```

## 7. 浏览器兼容性注意事项
```javascript
// 1. rangeCount 在某些旧浏览器中可能不存在
const rangeCount = selection.rangeCount || 
                   (selection.getRangeAt ? 1 : 0);

// 2. 多选区的浏览器差异
if (selection.rangeCount > 1) {
  // Firefox 支持多选区，其他浏览器通常不支持
  console.log('多个选区:', selection.rangeCount);
}

// 3. 类型检查的兼容写法
const type = selection.type || 
             (selection.isCollapsed ? 'Caret' : 'Range');

// 4. 安全获取 Range
function getSafeRange(selection) {
  try {
    return selection.rangeCount > 0 ? selection.getRangeAt(0) : null;
  } catch (e) {
    // 某些浏览器可能抛出异常
    return null;
  }
}
```

## 总结
+ `rangeCount`：最重要的属性，表示有多少个选区
+ `isCollapsed`：判断是光标还是文本选择
+ **Selection 包含 Range**：通过 `getRangeAt(index)` 获取具体的 Range 对象
+ **Range 有起止点**：`startContainer`/`startOffset` 和 `endContainer`/`endOffset`
+ **光标恢复的关键**：保存 Range 的起止节点和偏移量

理解这些 API 对于实现富文本编辑器、Markdown 编辑器等至关重要。掌握它们可以让你精确控制用户的输入和选择行为。
