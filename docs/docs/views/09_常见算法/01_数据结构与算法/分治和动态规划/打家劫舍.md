

## 打家劫舍问题：不能偷相邻的房子，求最大金额
```js
function robDP(houses) {
    console.log(`\n=== 打家劫舍问题：房子金额 ${houses} ===\n`);
    
    if (!houses || houses.length === 0) return 0;
    if (houses.length === 1) return houses[0];
    
    // 1. 定义状态：dp[i] 表示偷到第i个房子的最大金额
    const dp = new Array(houses.length);
    
    // 2. 初始化
    dp[0] = houses[0];  // 只有第一个房子
    dp[1] = Math.max(houses[0], houses[1]);  // 前两个房子选金额大的
    
    console.log(`只考虑房子0: 最大金额 = ${dp[0]}`);
    console.log(`考虑房子0-1: 最大金额 = max(${houses[0]}, ${houses[1]}) = ${dp[1]}\n`);
    
    // 3. 状态转移
    for (let i = 2; i < houses.length; i++) {
        // 关键：偷当前房子 vs 不偷当前房子
        const option1 = dp[i-1];                   // 不偷当前房子
        const option2 = dp[i-2] + houses[i];       // 偷当前房子
        
        dp[i] = Math.max(option1, option2);
        
        console.log(`考虑房子0-${i}:`);
        console.log(`  方案1: 不偷房子${i}，继承前${i-1}个的最优解 = ${option1}`);
        console.log(`  方案2: 偷房子${i}(${houses[i]}元) + 前${i-2}个的最优解(${dp[i-2]}) = ${option2}`);
        console.log(`  最优选择: max(${option1}, ${option2}) = ${dp[i]}\n`);
    }
    
    return dp[houses.length - 1];
}

// 测试
const houses = [2, 7, 9, 3, 1];
console.log(`最大可偷金额: ${robDP(houses)}`);
// 解释：偷房子0(2)、房子2(9)、房子4(1)，总共12
```
## 打家劫舍
```js
function rob(nums) {
    /*
    问题：不能偷相邻的房子，求最大金额
    
    状态定义：dp[i] 表示偷到第i个房子时的最大金额
    状态转移：dp[i] = max(dp[i-1], dp[i-2] + nums[i])
    初始化：dp[0] = nums[0], dp[1] = max(nums[0], nums[1])
    */
    if (!nums || nums.length === 0) return 0;
    if (nums.length === 1) return nums[0];
    
    const n = nums.length;
    const dp = new Array(n);
    
    dp[0] = nums[0];
    dp[1] = Math.max(nums[0], nums[1]);
    
    for (let i = 2; i < n; i++) {
        dp[i] = Math.max(dp[i - 1], dp[i - 2] + nums[i]);
    }
    
    return dp[n - 1];
}

// 优化版（滚动数组）
function robOptimized(nums) {
    if (!nums || nums.length === 0) return 0;
    if (nums.length === 1) return nums[0];
    
    let prev2 = 0;  // dp[i-2]
    let prev1 = 0;  // dp[i-1]
    
    for (let i = 0; i < nums.length; i++) {
        const curr = Math.max(prev1, prev2 + nums[i]);
        prev2 = prev1;
        prev1 = curr;
    }
    
    return prev1;
}

// 测试
const houses = [2, 7, 9, 3, 1];
console.log('打家劫舍:', rob(houses));  // 12 (2 + 9 + 1)
console.log('优化版:', robOptimized(houses));  // 12
```
