### 发工资 分治方法

```js
// 分治法发工资
function paySalariesDivideConquer(allEmployees) {
    console.log("我是财务总监，我要用分治法发工资！");
    
    // 1. 基线条件：如果只有1个员工，我自己处理
    if (allEmployees.length <= 1) {
        console.log(`直接处理 ${allEmployees[0].name} 的工资`);
        return "完成！";
    }
    
    console.log(`有 ${allEmployees.length} 个员工需要处理`);
    
    // 2. 分：把员工分成两组
    const mid = Math.floor(allEmployees.length / 2);
    const groupA = allEmployees.slice(0, mid);  // 前500人
    const groupB = allEmployees.slice(mid);     // 后500人
    
    console.log(`分成A组(${groupA.length}人)和B组(${groupB.length}人)`);
    
    // 3. 治：找两个经理分别处理
    console.log("叫来财务经理A处理A组...");
    const resultA = paySalariesDivideConquer(groupA);  // 经理A递归处理
    
    console.log("叫来财务经理B处理B组...");
    const resultB = paySalariesDivideConquer(groupB);  // 经理B递归处理
    
    // 4. 合：汇总结果
    console.log("两个经理都完成了，我只需要汇总结果！");
    return `A组:${resultA}, B组:${resultB}`;
}
```

```js
function sumDivideConquer(arr) {
    console.log(`计算数组 [${arr}] 的和`);
    
    // 1. 基线条件：如果数组长度很小，直接计算
    if (arr.length === 0) {
        console.log("空数组，和是 0");
        return 0;
    }
    
    if (arr.length === 1) {
        console.log(`只有一个数 ${arr[0]}，直接返回`);
        return arr[0];
    }
    
    // 2. 分：把数组分成两半
    const mid = Math.floor(arr.length / 2);
    const left = arr.slice(0, mid);   // 左半部分
    const right = arr.slice(mid);     // 右半部分
    
    console.log(`分成左半部分 [${left}] 和右半部分 [${right}]`);
    
    // 3. 治：递归计算两半的和
    console.log("递归计算左半部分的和...");
    const leftSum = sumDivideConquer(left);
    
    console.log("递归计算右半部分的和...");
    const rightSum = sumDivideConquer(right);
    
    // 4. 合：合并结果
    const total = leftSum + rightSum;
    console.log(`合并结果：${leftSum} + ${rightSum} = ${total}`);
    
    return total;
}

// 测试
const numbers = [1, 2, 3, 4, 5, 6, 7, 8];
console.log("分治法计算数组和:", sumDivideConquer(numbers));
```

### 归并排序（分治方法）
```js
// 想象你要整理一副乱序的扑克牌
function mergeSort(cards) {
    console.log(`整理这些扑克牌: [${cards}]`);
    
    // 1. 基线条件：如果只有0张或1张牌，已经有序
    if (cards.length <= 1) {
        console.log(`只有 ${cards.length} 张牌，不用整理了`);
        return cards;
    }
    
    // 2. 分：把牌堆分成两半
    const mid = Math.floor(cards.length / 2);
    const leftPile = cards.slice(0, mid);  // 左半堆
    const rightPile = cards.slice(mid);    // 右半堆
    
    console.log(`分成两堆: 左堆[${leftPile}] 和 右堆[${rightPile}]`);
    
    // 3. 治：分别整理两堆牌（递归调用自己）
    console.log("找人整理左堆...");
    const sortedLeft = mergeSort(leftPile);
    
    console.log("找人整理右堆...");
    const sortedRight = mergeSort(rightPile);
    
    // 4. 合：把两堆有序的牌合并成一堆有序的牌
    console.log(`合并两堆有序的牌: [${sortedLeft}] 和 [${sortedRight}]`);
    const merged = merge(sortedLeft, sortedRight);
    console.log(`合并结果: [${merged}]`);
    
    return merged;
}

function merge(left, right) {
    // 想象你有两堆已经排好序的牌，如何合并成一堆？
    const result = [];
    let i = 0, j = 0;
    
    // 比较两堆最上面的牌，取较小的那张
    while (i < left.length && j < right.length) {
        if (left[i] < right[j]) {
            result.push(left[i]);
            i++;
        } else {
            result.push(right[j]);
            j++;
        }
    }
    
    // 把剩下的牌直接放进去
    return [...result, ...left.slice(i), ...right.slice(j)];
}

// 测试：整理乱序的牌
const cards = [7, 2, 5, 1, 8, 3, 6, 4];
console.log("原始牌堆:", cards);
console.log("排序后:", mergeSort(cards));
```

```js 
function findMaxDivideConquer(arr, start = 0, end = arr.length - 1) {
    console.log(`在数组 arr[${start}...${end}] 中找最大值`);
    
    // 1. 基线条件：如果只有一个元素
    if (start === end) {
        console.log(`只有一个元素 ${arr[start]}，直接返回`);
        return arr[start];
    }
    
    // 2. 分：分成两半
    const mid = Math.floor((start + end) / 2);
    console.log(`分成两半: [${start}...${mid}] 和 [${mid+1}...${end}]`);
    
    // 3. 治：分别找两半的最大值
    console.log("找左半部分的最大值...");
    const leftMax = findMaxDivideConquer(arr, start, mid);
    
    console.log("找右半部分的最大值...");
    const rightMax = findMaxDivideConquer(arr, mid + 1, end);
    
    // 4. 合：取两者中较大的
    const result = Math.max(leftMax, rightMax);
    console.log(`合并结果: max(${leftMax}, ${rightMax}) = ${result}`);
    
    return result;
}
```

### 归并排序 分治
```js
function mergeSort(arr) {
    // 基线条件：数组长度为0或1时已有序
    if (arr.length <= 1) {
        return arr;
    }
    
    // 1. 分：分成两半
    const mid = Math.floor(arr.length / 2);
    const left = arr.slice(0, mid);
    const right = arr.slice(mid);
    
    // 2. 治：递归排序
    const sortedLeft = mergeSort(left);
    const sortedRight = mergeSort(right);
    
    // 3. 合：合并有序数组
    return merge(sortedLeft, sortedRight);
}

function merge(left, right) {
    const result = [];
    let i = 0, j = 0;
    
    while (i < left.length && j < right.length) {
        if (left[i] <= right[j]) {
            result.push(left[i]);
            i++;
        } else {
            result.push(right[j]);
            j++;
        }
    }
    
    // 合并剩余元素
    return result.concat(left.slice(i), right.slice(j));
}

// 测试
const arr = [38, 27, 43, 3, 9, 82, 10];
console.log('归并排序:', mergeSort(arr));
// 输出: [3, 9, 10, 27, 38, 43, 82]
```
 
### 快速排序 分治
```js
function quickSort(arr) {
    // 基线条件
    if (arr.length <= 1) {
        return arr;
    }
    
    // 选择基准值（这里选中间元素）
    const pivotIndex = Math.floor(arr.length / 2);
    const pivot = arr[pivotIndex];
    
    // 分：分成三部分
    const left = [];
    const middle = [];
    const right = [];
    
    for (let i = 0; i < arr.length; i++) {
        if (i === pivotIndex) continue;
        
        if (arr[i] < pivot) {
            left.push(arr[i]);
        } else if (arr[i] > pivot) {
            right.push(arr[i]);
        } else {
            middle.push(arr[i]);
        }
    }
    
    // 治 + 合：递归排序并合并
    return [...quickSort(left), pivot, ...quickSort(right)];
}

// 测试
const nums = [3, 6, 8, 10, 1, 2, 1];
console.log('快速排序:', quickSort(nums));
// 输出: [1, 1, 2, 3, 6, 8, 10]
```

### 二分查找 分治
```js
function binarySearch(arr, target, left = 0, right = arr.length - 1) {
    // 基线条件：搜索范围为空
    if (left > right) {
        return -1;
    }
    
    // 分：找到中间位置
    const mid = Math.floor((left + right) / 2);
    
    // 治：比较并决定搜索哪一边
    if (arr[mid] === target) {
        return mid;  // 找到
    } else if (arr[mid] > target) {
        // 搜索左半边
        return binarySearch(arr, target, left, mid - 1);
    } else {
        // 搜索右半边
        return binarySearch(arr, target, mid + 1, right);
    }
}

// 测试
const sortedArr = [1, 3, 5, 7, 9, 11, 13];
console.log('二分查找 7:', binarySearch(sortedArr, 7));  // 3
console.log('二分查找 12:', binarySearch(sortedArr, 12)); // -1
```

### 最大子数组和 分治
```js
function maxSubarraySumDivideConquer(nums) {
    if (!nums || nums.length === 0) return 0;
    
    function helper(left, right) {
        // 基线条件：只有一个元素
        if (left === right) {
            return nums[left];
        }
        
        // 分：找到中间位置
        const mid = Math.floor((left + right) / 2);
        
        // 治：递归求左右两边的最大值
        const leftMax = helper(left, mid);
        const rightMax = helper(mid + 1, right);
        
        // 合：求跨越中间的最大值
        const crossMax = maxCrossingSum(left, mid, right);
        
        // 返回三者中的最大值
        return Math.max(leftMax, rightMax, crossMax);
    }
    
    function maxCrossingSum(left, mid, right) {
        // 向左找最大和
        let leftSum = -Infinity;
        let sum = 0;
        for (let i = mid; i >= left; i--) {
            sum += nums[i];
            if (sum > leftSum) leftSum = sum;
        }
        
        // 向右找最大和
        let rightSum = -Infinity;
        sum = 0;
        for (let i = mid + 1; i <= right; i++) {
            sum += nums[i];
            if (sum > rightSum) rightSum = sum;
        }
        
        return leftSum + rightSum;
    }
    
    return helper(0, nums.length - 1);
}

// 测试
const nums = [-2, 1, -3, 4, -1, 2, 1, -5, 4];
console.log('分治法最大子数组和:', maxSubarraySumDivideConquer(nums)); // 6
```

### 分治模版

```js
// 分治算法万能模板
function divideAndConquer(problem) {
    // 第一步：检查是否可以直接解决（基线条件）
    if (problem is small enough) {
        return solveDirectly(problem);
    }
    
    // 第二步：分解问题
    const subproblems = divide(problem);
    
    // 第三步：递归解决子问题
    const solutions = [];
    for (let subproblem of subproblems) {
        solutions.push(divideAndConquer(subproblem));
    }
    
    // 第四步：合并子问题的解
    return combine(solutions);
}
```
