
## 存钱罐问题
- 问题：你有不同面值的硬币（1元、2元、5元），要用最少的硬币凑出11元。


### 存钱罐 动态规划方法
```js
function coinChangeDP(coins, amount) {
    console.log(`\n=== 动态规划：用硬币 [${coins}] 凑出 ${amount} 元 ===\n`);
    
    // 1. 定义状态：dp[i] 表示凑出金额 i 需要的最少硬币数
    const dp = new Array(amount + 1).fill(Infinity);
    
    // 2. 初始化：凑出0元需要0个硬币
    dp[0] = 0;
    console.log(`dp[0] = ${dp[0]} (凑出0元需要0个硬币)\n`);
    
    // 3. 从1元开始，计算到目标金额
    for (let i = 1; i <= amount; i++) {
        console.log(`计算凑出 ${i} 元的最少硬币数：`);
        
        // 尝试每种硬币
        for (let coin of coins) {
            if (i >= coin) {
                // 关键：凑出 i 元 = 先用掉一个 coin 元硬币 + 凑出剩下的 (i-coin) 元
                dp[i] = Math.min(dp[i], dp[i - coin] + 1);
                console.log(`  尝试用 ${coin} 元硬币：dp[${i}] = min(${dp[i]}, dp[${i - coin}]+1)`);
            }
        }
        
        console.log(`  结果：凑出 ${i} 元最少需要 ${dp[i]} 个硬币\n`);
    }
    
    return dp[amount] === Infinity ? -1 : dp[amount];
}

// 测试：用1元、2元、5元硬币凑11元
const coins = [1, 2, 5];
const amount = 11;
console.log(`最少需要 ${coinChangeDP(coins, amount)} 个硬币`);
// 答案是3个：5+5+1 或 5+2+2+2
```
