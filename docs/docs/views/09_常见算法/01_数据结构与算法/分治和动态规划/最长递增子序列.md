

## 最长递增子序列

```js
function lengthOfLIS(nums) {
    console.log(`\n=== 最长递增子序列：数组 ${nums} ===\n`);
    
    if (!nums || nums.length === 0) return 0;
    
    // 1. 定义状态：dp[i] 表示以 nums[i] 结尾的最长递增子序列长度
    const dp = new Array(nums.length).fill(1);
    
    console.log(`初始状态：每个元素自身就是一个子序列，长度=1`);
    console.log(`dp = [${dp}]\n`);
    
    let maxLen = 1;
    
    // 2. 状态转移
    for (let i = 1; i < nums.length; i++) {
        console.log(`处理 nums[${i}] = ${nums[i]}:`);
        
        for (let j = 0; j < i; j++) {
            if (nums[i] > nums[j]) {
                // 如果nums[i]可以接在nums[j]后面
                dp[i] = Math.max(dp[i], dp[j] + 1);
                console.log(`  nums[${i}](${nums[i]}) > nums[${j}](${nums[j]})，可以接在后面`);
                console.log(`  dp[${i}]更新为 max(${dp[i]}, dp[${j}]+1=${dp[j]}+1=${dp[j]+1}) = ${dp[i]}`);
            }
        }
        
        maxLen = Math.max(maxLen, dp[i]);
        console.log(`  当前最长递增子序列长度: ${maxLen}\n`);
    }
    
    return maxLen;
}

// 测试
const nums = [10, 9, 2, 5, 3, 7, 101, 18];
console.log(`最长递增子序列长度: ${lengthOfLIS(nums)}`);
// 解释：[2, 5, 7, 101] 或 [2, 5, 7, 18]，长度都是4
```

```js
function lengthOfLIS(nums) {
    /*
    问题：求最长递增子序列的长度
    
    状态定义：dp[i] 表示以nums[i]结尾的最长递增子序列长度
    状态转移：dp[i] = max(dp[j]) + 1, 其中 0 <= j < i 且 nums[j] < nums[i]
    初始化：所有dp[i] = 1
    */
    if (!nums || nums.length === 0) return 0;
    
    const n = nums.length;
    const dp = new Array(n).fill(1);  // 每个元素本身就是一个子序列
    
    let maxLen = 1;
    
    for (let i = 1; i < n; i++) {
        for (let j = 0; j < i; j++) {
            if (nums[i] > nums[j]) {
                dp[i] = Math.max(dp[i], dp[j] + 1);
            }
        }
        maxLen = Math.max(maxLen, dp[i]);
    }
    
    return maxLen;
}

// 测试
const sequence = [10, 9, 2, 5, 3, 7, 101, 18];
console.log('最长递增子序列长度:', lengthOfLIS(sequence));  // 4
// 解释：最长递增子序列是 [2, 5, 7, 101] 或 [2, 5, 7, 18]
```
