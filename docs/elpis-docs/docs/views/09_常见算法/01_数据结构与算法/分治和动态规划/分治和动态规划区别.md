分治	大事化小，各个击破	管理团队：把大任务分解，分给不同人做
动态规划	记住教训，避免重复	聪明学习：错题本记下错误，下次不再犯

function chooseAlgorithm(problem) {
    // 1. 问题是否有最优子结构？
    //    - 大问题的最优解包含小问题的最优解
    
    // 2. 子问题是否重叠？
    if (hasOverlappingSubproblems(problem)) {
        // 使用动态规划
        return '动态规划';
    } else {
        // 3. 问题能否分解？
        if (canBeDivided(problem)) {
            // 使用分治算法
            return '分治算法';
        } else {
            // 使用其他算法
            return '其他算法';
        }
    }
}


// 示例1：最大子数组和
function maxSubarraySum(nums) {
    // 方法1：暴力法 - O(n²)
    // 方法2：分治法 - O(n log n)  ← 子问题不重叠
    // 方法3：动态规划 - O(n)       ← 更优的选择
    // 方法4：贪心法 - O(n)
    
    // 实际中常用动态规划：
    let maxSum = nums[0];
    let currentSum = nums[0];
    
    for (let i = 1; i < nums.length; i++) {
        currentSum = Math.max(nums[i], currentSum + nums[i]);
        maxSum = Math.max(maxSum, currentSum);
    }
    
    return maxSum;
}
