# 斐波那契数列

## 问题：求第n个斐波那契数（F(0)=0, F(1)=1, F(n)=F(n-1)+F(n-2)）



## 斐波那契数列 分治算法
```js
// 1. 分治解法（笨方法）
function fibonacciDivideConquer(n) {
    console.count(`计算 F(${n})`);
    
    if (n <= 1) return n;
    
    // 问题：大量重复计算！
    return fibonacciDivideConquer(n-1) + fibonacciDivideConquer(n-2);
}

```
## 斐波那契数列 动态规划解法
```js
// 2. 动态规划解法（聪明方法）
function fibonacciDP(n) {
    if (n <= 1) return n;
    
    // 创建记忆数组
    const dp = new Array(n + 1);
    dp[0] = 0;
    dp[1] = 1;
    
    console.log(`从 F(2) 开始计算...`);
    
    // 从小到大计算
    for (let i = 2; i <= n; i++) {
        dp[i] = dp[i-1] + dp[i-2];
        console.log(`F(${i}) = F(${i-1}) + F(${i-2}) = ${dp[i-1]} + ${dp[i-2]} = ${dp[i]}`);
    }
    
    return dp[n];
}
// 对比测试
console.log("=== 分治算法（大量重复计算）===");
console.log("F(5) =", fibonacciDivideConquer(5));
console.log(`计算次数: ${console.count(`计算 F`)} 次\n`);

console.log("=== 动态规划（高效计算）===");
console.log("F(5) =", fibonacciDP(5));
```
### 动态规划的两种实现方式
- 自底向上（Bottom-Up）
```js
// 从小到大计算，从基础情况开始
function bottomUp(n) {
    // 1. 创建dp表
    const dp = new Array(n + 1);
    
    // 2. 初始化最简单的情况
    dp[0] = 基础值;
    dp[1] = 基础值;
    
    // 3. 从小问题逐渐解决大问题
    for (let i = 2; i <= n; i++) {
        dp[i] = 根据 dp[i-1]、dp[i-2]... 计算;
    }
    
    return dp[n];
}
```

- 自顶向下（Top-Down）

```js
// 也叫"记忆化搜索"，从目标开始，用备忘录记录
function topDown(n, memo = {}) {
    // 1. 如果已经计算过，直接返回
    if (n in memo) return memo[n];
    
    // 2. 基础情况
    if (n === 0) return 基础值;
    if (n === 1) return 基础值;
    
    // 3. 计算并保存到备忘录
    memo[n] = topDown(n-1, memo) + topDown(n-2, memo);
    
    return memo[n];
}
```

```js
// ❌ 暴力递归 - O(2^n)
function fibonacciRecursive(n) {
    if (n <= 1) return n;
    return fibonacciRecursive(n - 1) + fibonacciRecursive(n - 2);
}

// ✅ 动态规划 - O(n)
function fibonacciDP(n) {
    if (n <= 1) return n;
    
    // 1. 定义状态：dp[i] 表示第i个斐波那契数
    const dp = new Array(n + 1);
    
    // 2. 初始化
    dp[0] = 0;
    dp[1] = 1;
    
    // 3. 状态转移
    for (let i = 2; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    
    return dp[n];
}

// ✅ 优化空间 - O(1)
function fibonacciOptimized(n) {
    if (n <= 1) return n;
    
    let prev = 0;  // dp[i-2]
    let curr = 1;  // dp[i-1]
    
    for (let i = 2; i <= n; i++) {
        const next = prev + curr;
        prev = curr;
        curr = next;
    }
    
    return curr;
}

// 测试
console.log('DP 斐波那契(10):', fibonacciDP(10));      // 55
console.log('优化 斐波那契(10):', fibonacciOptimized(10)); // 55
```
