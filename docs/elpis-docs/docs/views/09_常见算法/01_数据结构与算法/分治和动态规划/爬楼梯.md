## 爬楼梯问题
- 你要爬楼梯，每次可以走1阶或2阶，爬到第n阶有多少种方法？


### 爬楼梯 分治算法
```js
// ❌ 笨方法：每次都重新计算
function climbStairsNaive(n) {
    console.log(`计算爬到第${n}阶的方法数`);
    
    if (n === 1) return 1;  // 只有1种方法：走1步
    if (n === 2) return 2;  // 2种方法：1+1 或 直接2步
    
    // 问题：这里会重复计算很多次！
    return climbStairsNaive(n-1) + climbStairsNaive(n-2);
}

// 试试计算爬到第5阶
console.log("暴力递归计算过程：");
console.log("爬到第5阶的方法数:", climbStairsNaive(5));

/*
计算爬到第5阶
    计算爬到第4阶
        计算爬到第3阶
            计算爬到第2阶 ✓
            计算爬到第1阶 ✓
        计算爬到第2阶 ✓（这里重新计算了一次！）
    计算爬到第3阶
        计算爬到第2阶 ✓（这里又计算了一次！）
        计算爬到第1阶 ✓
你看，爬到第3阶计算了2次，爬到第2阶计算了3次！这就是重复计算的问题。
*/
```
### 爬楼梯 动态规划方法
```js
// ✅ 聪明方法：记住已经算过的结果
function climbStairsDP(n) {
    console.log(`=== 动态规划计算爬到第${n}阶 ===\n`);
    
    if (n <= 2) return n;
    
    // 1. 创建一个"记忆本子"（数组），记录已经算过的结果
    const memo = new Array(n + 1).fill(0);
    
    // 2. 记下最简单的情况（基础事实）
    memo[1] = 1;  // 爬到第1阶：1种方法
    memo[2] = 2;  // 爬到第2阶：2种方法
    
    console.log(`已知：爬到第1阶有 ${memo[1]} 种方法`);
    console.log(`已知：爬到第2阶有 ${memo[2]} 种方法\n`);
    
    // 3. 从第3阶开始，利用已知结果计算未知结果
    for (let i = 3; i <= n; i++) {
        // 关键思想：爬到第i阶 = 从第i-1阶走1步 + 从第i-2阶走2步
        memo[i] = memo[i-1] + memo[i-2];
        console.log(`爬到第${i}阶 = 爬到第${i-1}阶(${memo[i-1]}种) + 爬到第${i-2}阶(${memo[i-2]}种) = ${memo[i]}种`);
    }
    
    console.log(`\n最终结果：爬到第${n}阶有 ${memo[n]} 种方法`);
    return memo[n];
}

// 测试
console.log(climbStairsDP(5));
```
### 爬楼梯
```js
function climbStairs(n) {
    /*
    问题：每次可以爬1或2阶，到第n阶有多少种方法？
    
    状态定义：dp[i] 表示到第i阶的方法数
    状态转移：dp[i] = dp[i-1] + dp[i-2]  (可以从i-1走1步，或从i-2走2步)
    初始化：dp[0] = 1, dp[1] = 1
    */
    if (n <= 1) return 1;
    
    const dp = new Array(n + 1);
    dp[0] = 1;
    dp[1] = 1;
    
    for (let i = 2; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    
    return dp[n];
}

// 优化版
function climbStairsOptimized(n) {
    if (n <= 1) return 1;
    
    let prev = 1;  // dp[i-2]
    let curr = 1;  // dp[i-1]
    
    for (let i = 2; i <= n; i++) {
        const next = prev + curr;
        prev = curr;
        curr = next;
    }
    
    return curr;
}

// 测试
console.log('爬楼梯(5):', climbStairs(5));  // 8
console.log('爬楼梯(10):', climbStairs(10)); // 89
```
