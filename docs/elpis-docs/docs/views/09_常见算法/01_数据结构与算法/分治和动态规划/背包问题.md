## 背包问题
```js
function knapsack01(weights, values, capacity) {
    /*
    0-1背包问题
    
    状态定义：dp[i][w] 表示前i个物品，容量为w时的最大价值
    状态转移：
        - 不放第i个物品：dp[i][w] = dp[i-1][w]
        - 放第i个物品：dp[i][w] = dp[i-1][w-weights[i-1]] + values[i-1]
        取两者最大值
    */
    const n = weights.length;
    // dp[i][w]
    const dp = Array.from({ length: n + 1 }, () => new Array(capacity + 1).fill(0));
    
    for (let i = 1; i <= n; i++) {
        for (let w = 1; w <= capacity; w++) {
            if (weights[i - 1] <= w) {
                // 可以选择放或不放
                dp[i][w] = Math.max(
                    dp[i - 1][w],  // 不放
                    dp[i - 1][w - weights[i - 1]] + values[i - 1]  // 放
                );
            } else {
                // 放不下
                dp[i][w] = dp[i - 1][w];
            }
        }
    }
    
    return dp[n][capacity];
}

// 优化版（一维数组）
function knapsack01Optimized(weights, values, capacity) {
    const n = weights.length;
    const dp = new Array(capacity + 1).fill(0);
    
    for (let i = 0; i < n; i++) {
        // 必须从后往前遍历，避免重复使用物品
        for (let w = capacity; w >= weights[i]; w--) {
            dp[w] = Math.max(dp[w], dp[w - weights[i]] + values[i]);
        }
    }
    
    return dp[capacity];
}

// 测试
const weights = [2, 3, 4, 5];
const values = [3, 4, 5, 6];
const capacity = 8;

console.log('0-1背包最大价值:', knapsack01(weights, values, capacity));  // 10
console.log('优化版:', knapsack01Optimized(weights, values, capacity));  // 10
// 解释：选择物品1(2,3)和物品4(5,6)，总重量7，价值9；或者物品2(3,4)和物品3(4,5)，总重量7，价值9
```
