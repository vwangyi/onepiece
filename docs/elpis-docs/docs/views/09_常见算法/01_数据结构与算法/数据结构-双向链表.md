# 双向链表
- https://leetcode.cn/problems/design-linked-list/description/ 链表初始化

// 双向链表节点类
class ListNode {
    val: number;
    next: ListNode | null;
    prev: ListNode | null;

    constructor(val?: number, next?: ListNode | null, prev?: ListNode | null) {
        this.val = val === undefined ? 0 : val;
        this.next = next === undefined ? null : next;
        this.prev = prev === undefined ? null : prev;
    }
}

// 双向链表类
class MyLinkedList {
    private size: number;
    private head: ListNode; // 虚拟头节点
    private tail: ListNode; // 虚拟尾节点

    constructor() {
        this.size = 0;
        this.head = new ListNode(0);
        this.tail = new ListNode(0);
        this.head.next = this.tail;
        this.tail.prev = this.head;
    }

    // 获取链表中第 index 个节点的值
    get(index: number): number {
        if (index < 0 || index >= this.size) {
            return -1;
        }
        
        let curr: ListNode;
        
        // 优化：根据索引位置决定从头还是从尾开始遍历
        if (index + 1 < this.size - index) {
            // 更靠近头部，从头开始遍历
            curr = this.head;
            for (let i = 0; i <= index; i++) {
                curr = curr.next!;
            }
        } else {
            // 更靠近尾部，从尾开始遍历
            curr = this.tail;
            for (let i = 0; i < this.size - index; i++) {
                curr = curr.prev!;
            }
        }
        return curr.val;
    }

    // 在链表头部添加节点
    addAtHead(val: number): void {
        this.addAtIndex(0, val);
    }

    // 在链表尾部添加节点
    addAtTail(val: number): void {
        this.addAtIndex(this.size, val);
    }

    // 在链表中第 index 个节点前添加节点
    addAtIndex(index: number, val: number): void {
        if (index > this.size) {
            return;
        }
        
        index = Math.max(0, index);
        
        let pred: ListNode, succ: ListNode;
        
        // 优化：决定是从头找还是从尾找
        if (index < this.size - index) {
            // 从头部开始找
            pred = this.head;
            for (let i = 0; i < index; i++) {
                pred = pred.next!;
            }
            succ = pred.next!;
        } else {
            // 从尾部开始找
            succ = this.tail;
            for (let i = 0; i < this.size - index; i++) {
                succ = succ.prev!;
            }
            pred = succ.prev!;
        }
        
        this.size++;
        const toAdd = new ListNode(val);
        
        // 建立双向连接
        toAdd.prev = pred;
        toAdd.next = succ;
        pred.next = toAdd;
        succ.prev = toAdd;
    }

    // 删除链表中第 index 个节点
    deleteAtIndex(index: number): void {
        if (index < 0 || index >= this.size) {
            return;
        }
        
        let pred: ListNode, succ: ListNode;
        
        if (index < this.size - index) {
            // 从头部开始找
            pred = this.head;
            for (let i = 0; i < index; i++) {
                pred = pred.next!;
            }
            succ = pred.next!.next!;
        } else {
            // 从尾部开始找
            succ = this.tail;
            for (let i = 0; i < this.size - index - 1; i++) {
                succ = succ.prev!;
            }
            pred = succ.prev!.prev!;
        }
        
        this.size--;
        pred.next = succ;
        succ.prev = pred;
    }

    // 返回链表正向字符串表示
    toString(): string {
        const values: number[] = [];
        let current: ListNode | null = this.head.next;
        
        while (current !== null && current !== this.tail) {
            values.push(current.val);
            current = current.next;
        }
        
        return `MyLinkedList: [${values.join(' ⇆ ')}] (size: ${this.size})`;
    }

    // 返回链表反向字符串表示（测试双向链表功能）
    toStringReverse(): string {
        const values: number[] = [];
        let current: ListNode | null = this.tail.prev;
        
        while (current !== null && current !== this.head) {
            values.push(current.val);
            current = current.prev;
        }
        
        return `MyLinkedList(反向): [${values.join(' ⇆ ')}] (size: ${this.size})`;
    }

    // 获取链表大小（可选方法）
    getSize(): number {
        return this.size;
    }

    // 判断链表是否为空（可选方法）
    isEmpty(): boolean {
        return this.size === 0;
    }
}

// 测试代码
function testMyLinkedList() {
    console.log('=== ES6 class 版本双向链表测试 ===');

    // 创建链表
    const linkedList = new MyLinkedList();
    console.log('初始链表:', linkedList.toString());
    console.log('链表是否为空:', linkedList.isEmpty());

    // 头部添加
    linkedList.addAtHead(1);
    console.log('头部添加1后:', linkedList.toString());

    // 尾部添加
    linkedList.addAtTail(3);
    console.log('尾部添加3后:', linkedList.toString());

    // 在中间添加
    linkedList.addAtIndex(1, 2);
    console.log('在索引1添加2后:', linkedList.toString());

    // 测试双向遍历
    console.log('正向遍历:', linkedList.toString());
    console.log('反向遍历:', linkedList.toStringReverse());

    // 获取节点值
    console.log('获取索引0:', linkedList.get(0)); // 1
    console.log('获取索引1:', linkedList.get(1)); // 2
    console.log('获取索引2:', linkedList.get(2)); // 3
    console.log('获取无效索引:', linkedList.get(5)); // -1

    // 删除节点
    linkedList.deleteAtIndex(1);
    console.log('删除索引1后:', linkedList.toString());
    console.log('删除后反向遍历:', linkedList.toStringReverse());

    // 测试边界情况
    linkedList.addAtHead(0);
    linkedList.addAtTail(4);
    console.log('头部加0，尾部加4后:', linkedList.toString());
    console.log('获取链表大小:', linkedList.getSize());

    // 测试删除尾部
    linkedList.deleteAtIndex(linkedList.getSize() - 1);
    console.log('删除尾部节点后:', linkedList.toString());

    // 测试从头部删除
    linkedList.deleteAtIndex(0);
    console.log('删除头部节点后:', linkedList.toString());
    console.log('链表是否为空:', linkedList.isEmpty());

    // 复杂测试
    console.log('\n=== 复杂测试 ===');
    const list2 = new MyLinkedList();
    list2.addAtTail(1);
    list2.addAtTail(2);
    list2.addAtTail(3);
    list2.addAtTail(4);
    list2.addAtTail(5);
    console.log('初始化5个节点:', list2.toString());

    list2.addAtIndex(2, 99);
    console.log('在索引2添加99后:', list2.toString());
    console.log('反向验证:', list2.toStringReverse());

    // 测试遍历优化
    console.log('\n=== 测试遍历优化 ===');
    console.log('获取索引0（从头部找）:', list2.get(0));
    console.log('获取索引4（从尾部找）:', list2.get(4));

    // 测试批量操作
    console.log('\n=== 批量操作测试 ===');
    const list3 = new MyLinkedList();
    for (let i = 0; i < 10; i++) {
        list3.addAtTail(i);
    }
    console.log('添加10个节点:', list3.toString());
    
    // 删除偶数索引的节点
    for (let i = list3.getSize() - 1; i >= 0; i -= 2) {
        list3.deleteAtIndex(i);
    }
    console.log('删除偶数索引后:', list3.toString());
}

// 执行测试
testMyLinkedList();
